; Reads one digit character, echoes it, converts ASCII to numeric value,
; stores the numeric value in memory, then halts.

                LOC 20

000024  142000  IN 0,0 ;read one character into R0
000025  144001  OUT 0,1 ;echo the character to printer
000026  012037  SMR 0,0,31 ;R0 <- R0 - MEM[31]  (subtract ASCII '0' = 48)
000027  004036  STR 0,0,30 ;store numeric value into MEM[30]
000030  000000  HLT

                LOC 30
000036  000000  Data 0 ;result storage
000037  000060  Data 48 ;ASCII value for '0'
